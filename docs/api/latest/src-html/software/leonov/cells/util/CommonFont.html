001package software.leonov.cells.util;
002
003import org.apache.poi.ss.usermodel.Font;
004
005/**
006 * Cross platform fonts that should work on all versions of Microsoft Excel.
007 * 
008 * @author Zheny Leonov
009 */
010public enum CommonFont {
011
012    /**
013     * Arial font.
014     */
015    ARIAL("Arial"),
016
017    /**
018     * Comic Sans MS font.
019     */
020    COMIC_SANS_MS("Comic Sans MS"),
021
022    /**
023     * Courier New font. This fond is monospaced.
024     */
025    COURIER_NEW("Courier New"),
026
027    /**
028     * Georgia font.
029     */
030    GEORGIA("Georgia"),
031
032    /**
033     * Times New Roman font.
034     */
035    TIMES_NEW_ROMAN("Times New Roman"),
036
037    /**
038     * Verdana font.
039     */
040    VERDANA("Verdana");
041
042    private final String name;
043
044    private CommonFont(final String name) {
045        this.name = name;
046    }
047
048    /**
049     * Returns the value of this {@code CommonFont} compatible with {@link Font#setFontName(String)}.
050     * 
051     * @return the value of this {@code CommonFont} compatible with {@link Font#setFontName(String)}.
052     */
053    public String getFontName() {
054        return name;
055    }
056
057}
