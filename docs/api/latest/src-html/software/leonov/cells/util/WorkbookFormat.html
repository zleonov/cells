001package software.leonov.cells.util;
002
003import java.nio.file.Path;
004
005import org.apache.poi.ss.usermodel.Workbook;
006import org.apache.poi.xssf.streaming.SXSSFWorkbook;
007import org.apache.poi.xssf.usermodel.XSSFWorkbook;
008
009import software.leonov.cells.Workbooks;
010
011/**
012 * Specifies the Microsoft Excel file format.
013 */
014public enum WorkbookFormat {
015
016    /**
017     * The <a target="_blank" href= "http://en.wikipedia.org/wiki/Microsoft_Excel#Binary">Excel Binary File Format</a>
018     * supported since Microsoft Office 2003.
019     */
020    BINARY("xls"),
021
022    /**
023     * The <a target="_blank" href= "https://en.wikipedia.org/wiki/Office_Open_XML">Office Open XML</a> format supported
024     * starting with Microsoft Office 2007.
025     */
026    OFFICE_OPEN_XML("xlsx"),
027
028    /**
029     * The streaming version of {@link #OFFICE_OPEN_XML Office Open XML}. See {@link SXSSFWorkbook} and
030     * {@link SXSSFWorkbook#SXSSFWorkbook(XSSFWorkbook) new SXSSFWorkbook(XSSFWorkbook)} for more information.
031     * <p>
032     * <b>Using this format is error prone and not recommended for developers who are making casual use of the API.</b>
033     * <p>
034     * <b>Note:</b> Allocated temporary files must be cleaned up explicitly by calling the {@link #dispose(Workbook)}.
035     * <p>
036     * See the {@link Workbooks#newWorkbook(WorkbookFormat) newWorkbook(WorkbookFormat)} and
037     * {@link Workbooks#open(Path, WorkbookFormat) open(Path, WorkbookFormat)} methods for further details.
038     */
039    STREAMING_OFFICE_OPEN_XML("xlsx");
040
041    final String extension;
042
043    private WorkbookFormat(final String extension) {
044        this.extension = extension;
045    }
046
047    public String getExtension() {
048        return extension;
049    }
050
051}
