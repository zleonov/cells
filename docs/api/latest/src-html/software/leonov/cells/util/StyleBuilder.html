015/**
016 * A builder for creating {@link CellStyle}s.
017 * <p>
018 * Example:
019 * <p>
020 * 
021 * <pre>
022 *   import static org.apache.poi.ss.usermodel.IndexedColors.*;
023 *   import static org.apache.poi.ss.usermodel.CellStyle.*;
024 *
025 *   // Creating a new style
026 *
027 *   final StyleBuilder builder = new StyleBuilder()
028 *                         .setWrapText(true)
029 *                         .setBorder(BorderStyle.DASH_DOT);
030 *
031 *   final CellStyle wrappedDashDotStyle = builder.create(workbook);
032 *
033 *   final CellStyle wrappedBorderThinStyle = builder
034 *                        .setBorder(BorderStyle.BORDER_THIN)
035 *                        .create(workbook);
036 *                        
037 *   // Update an existing style
038 *   
039 *   CellStyle style = workbook.createCellStyle();
040 *   ...
041 *   new StyleBuilder().setWrapText(false).update(style);
042 * </pre>
043 * 
044 * Builder instances are reusable. It maintains its own state and can create or update multiple {@code CellStyle}
045 * instances across different workbooks.
046 * <p>
047 * <b>Note:</b> A workbook can store a finite number of cell-styles. Be careful not to create identical instances.
048 * Styles should be reused whenever possible.
049 * 
050 * @author Zhenya Leonov
051 */
052public final class StyleBuilder {

090    /**

092     */
093    public StyleBuilder() {

096    /**
097     * Creates a new {@code CellStyle} in the provided workbook based on the current builder settings.
098     * 
099     * @param workbook the workbook where the cell-style will be created
100     * @return a newly-created {@code CellStyle}
101     */
102    public CellStyle create(final Workbook workbook) {
103        checkNotNull(workbook, "workbook == null");
104        return update(workbook.createCellStyle());
105    }
106
107    /**
108     * Creates a new {@code CellStyle} in the provided workbook, initialized with the specified cell-style, and then updated
109     * with the current builder settings.
110     * 
111     * @param workbook  the workbook where the cell-style will be created
112     * @param baseStyle the cell-style to use as a base
113     * @return the newly-created {@code CellStyle}
114     */
115    public CellStyle create(final Workbook workbook, final CellStyle baseStyle) {
116        checkNotNull(workbook, "workbook == null");
117        checkNotNull(baseStyle, "baseStyle == null");
118
119        final CellStyle style = workbook.createCellStyle();
120        style.cloneStyleFrom(baseStyle);
121        return update(style);
122    }
123
124    /**
125     * Updates the provided style with the current builder settings.
126     * 
127     * @param style the style to update
128     * @return the updated style
129     */
130    public CellStyle update(final CellStyle style) {
131        checkNotNull(style, "style == null");
132        applyToStyle(style);
133        return style;
134    }

162        this.verticalAlignment   = null;

179    /**

181     * 
182     * @param border the border style

184     */
185    public StyleBuilder setBorder(final BorderStyle border) {
186        checkNotNull(border, "border == null");
187        this.topBorder    = border;
188        this.bottomBorder = border;
189        this.leftBorder   = border;
190        this.rightBorder  = border;
191        return this;
192    }

194    /**
195     * Sets the type of border to use for the specified {@code BorderSide} of the cell.
196     * 
197     * @param side   which side of the cell to set
198     * @param border the border style
199     * @return this {@code StyleBuilder} instance
200     */
201    public StyleBuilder setBorder(final BorderSide side, final BorderStyle border) {
202        checkNotNull(side, "side == null");
203        checkNotNull(border, "border == null");
204
205        if (side == BorderSide.TOP)
206            this.topBorder = border;
207        else if (side == BorderSide.BOTTOM)
208            this.bottomBorder = border;
209        else if (side == BorderSide.LEFT)
210            this.leftBorder = border;
211        else
212            this.rightBorder = border;
213
214        return this;
215    }
216
217    /**
218     * Sets the color to use for the border of the entire cell (top/right/bottom/left).
219     * 
220     * @param color the color to set
221     * @return this {@code StyleBuilder} instance
222     */
223    public StyleBuilder setBorderColor(final IndexedColors color) {
224        checkNotNull(color, "color == null");
225
226        for (final BorderSide side : BorderSide.values())
227            setBorderColor(side, color);
228
229        return this;
230    }

442    private void applyToStyle(final CellStyle style) {
443        if (horizontalAlignment != null)
444            style.setAlignment(horizontalAlignment);
445
446        if (topBorder != null)
447            style.setBorderTop(topBorder);
448        if (bottomBorder != null)
449            style.setBorderBottom(bottomBorder);
450        if (leftBorder != null)
451            style.setBorderLeft(leftBorder);
452        if (rightBorder != null)
453            style.setBorderRight(rightBorder);
454
455        // Apply border colors
456        if (topBorderColor != null)
457            applyBorderColor(style, BorderSide.TOP, topBorderColor);
458        if (bottomBorderColor != null)
459            applyBorderColor(style, BorderSide.BOTTOM, bottomBorderColor);
460        if (leftBorderColor != null)
461            applyBorderColor(style, BorderSide.LEFT, leftBorderColor);
462        if (rightBorderColor != null)
463            applyBorderColor(style, BorderSide.RIGHT, rightBorderColor);
464
465        if (dataFormat != null)
466            style.setDataFormat(dataFormat);
467
468        // Apply fill styles
469        if (fillForegroundColor != null)
470            applyFillForegroundColor(style, fillForegroundColor);
471        if (fillBackgroundColor != null)
472            applyFillBackgroundColor(style, fillBackgroundColor);
473        if (fillPattern != null)
474            style.setFillPattern(fillPattern);
475
476        if (font != null)
477            style.setFont(font);
478        if (hidden != null)
479            style.setHidden(hidden);
480        if (indention != null)
481            style.setIndention(indention);
482        if (locked != null)
483            style.setLocked(locked);
484        if (quotePrefixed != null)
485            style.setQuotePrefixed(quotePrefixed);
486        if (rotation != null)
487            style.setRotation(rotation);
488        if (shrinkToFit != null)
489            style.setShrinkToFit(shrinkToFit);
490        if (verticalAlignment != null)
491            style.setVerticalAlignment(verticalAlignment);
492        if (wrapText != null)
493            style.setWrapText(wrapText);
494    }

496    /**
497     * Returns a new {@code StyleBuilder} instance populated with the current properties of {@code this}
498     * {@code StyleBuilder}.
499     * 
500     * @return a new {@code StyleBuilder} instance populated with the current properties of {@code this}
501     *         {@code StyleBuilder}
502     */
503    public StyleBuilder newStyleBuilder() {
504        final StyleBuilder builder = new StyleBuilder();
505
506        // Border styles
507        builder.topBorder    = topBorder;
508        builder.bottomBorder = bottomBorder;
509        builder.leftBorder   = leftBorder;
510        builder.rightBorder  = rightBorder;
511
512        // Border colors
513        builder.topBorderColor    = topBorderColor;
514        builder.bottomBorderColor = bottomBorderColor;
515        builder.leftBorderColor   = leftBorderColor;
516        builder.rightBorderColor  = rightBorderColor;
517
518        // Data format
519        builder.dataFormat = dataFormat;
520
521        // Fill colors and pattern
522        builder.fillBackgroundColor = fillBackgroundColor;
523        builder.fillForegroundColor = fillForegroundColor;
524        builder.fillPattern         = fillPattern;
525
526        // Font
527        builder.font = font;
528
529        // Cell alignment
530        builder.horizontalAlignment = horizontalAlignment;
531        builder.verticalAlignment   = verticalAlignment;
532
533        // Cell properties
534        builder.hidden        = hidden;
535        builder.indention     = indention;
536        builder.locked        = locked;
537        builder.quotePrefixed = quotePrefixed;
538        builder.rotation      = rotation;
539        builder.shrinkToFit   = shrinkToFit;
540        builder.wrapText      = wrapText;
541
542        return builder;
543    }

545    private void applyBorderColor(final CellStyle style, final BorderSide side, final IndexedColors color) {
546        if (color == null)
547            return;
548
549        style.setTopBorderColor(color.getIndex());
550        style.setBottomBorderColor(color.getIndex());
551        style.setLeftBorderColor(color.getIndex());
552        style.setRightBorderColor(color.getIndex());
553    }
554
555    private void applyFillForegroundColor(final CellStyle style, final IndexedColors color) {
556        style.setFillForegroundColor(color.getIndex());
557    }
558
559    private void applyFillBackgroundColor(final CellStyle style, final IndexedColors color) {
560        style.setFillBackgroundColor(color.getIndex());
561    }
